61 bool connectNetwork(const char* apn, const char* user, const char* pass)
62 {
63 BLYNK_LOG1(BLYNK_F("Modem init..."));
64 if (!modem->begin()) {
65 BLYNK_FATAL(BLYNK_F("Cannot init"));
66 }
67
68 switch (modem->getSimStatus()) {
69 case SIM_ERROR: BLYNK_FATAL(BLYNK_F("SIM is missing")); break;
70 case SIM_LOCKED: BLYNK_FATAL(BLYNK_F("SIM is PIN-locked")); break;
71 default: break;
72 }
73
74 BLYNK_LOG1(BLYNK_F("Connecting to network..."));
75 if (modem->waitForNetwork()) {
76 String op = modem->getOperator();
77 BLYNK_LOG2(BLYNK_F("Network: "), op);
78 } else {
79 BLYNK_FATAL(BLYNK_F("Register in network failed"));
80 }
81
82 BLYNK_LOG3(BLYNK_F("Connecting to "), apn, BLYNK_F(" ..."));
83 if (!modem->gprsConnect(apn, user, pass)) {
84 BLYNK_FATAL(BLYNK_F("Connect GPRS failed"));
85 return false;
86 }
87
88 BLYNK_LOG1(BLYNK_F("Connected to GPRS"));
89 return true;
90 }
