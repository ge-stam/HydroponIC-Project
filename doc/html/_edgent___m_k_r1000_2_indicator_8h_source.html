45 // Reset counter if indicator state changes

47 m_PrevState = currState;

51 if (g_buttonPressed) {
52 if (millis() - g_buttonPressTime > BUTTON_HOLD_TIME_ACTION) { return beatLED(COLOR_WHITE, (int[]){ 100, 100 }); }
53 if (millis() - g_buttonPressTime > BUTTON_HOLD_TIME_INDICATION) { return waveLED(COLOR_WHITE, 1000); }
54 }
55 switch (currState) {
56 case MODE_RESET_CONFIG:
57 case MODE_WAIT_CONFIG: return beatLED(COLOR_BLUE, (int[]){ 50, 500 });
58 case MODE_CONFIGURING: return beatLED(COLOR_BLUE, (int[]){ 200, 200 });
59 case MODE_CONNECTING_NET: return beatLED(COLOR_BLYNK, (int[]){ 50, 500 });
60 case MODE_CONNECTING_CLOUD: return beatLED(COLOR_BLYNK, (int[]){ 100, 100 });
61 case MODE_RUNNING: return waveLED(COLOR_BLYNK, 5000);
62 case MODE_OTA_UPGRADE: return beatLED(COLOR_MAGENTA, (int[]){ 50, 50 });

144 template<typename T>
145 uint32_t beatLED(uint32_t onColor, const T& beat) {
146 const uint8_t cnt = sizeof(beat)/sizeof(beat[0]);
147 setRGB((m_Counter % 2 == 0) ? onColor : (uint32_t)COLOR_BLACK);
148 uint32_t next = beat[m_Counter % cnt];
149 m_Counter = (m_Counter+1) % cnt;
150 return next;
151 }

153 uint32_t waveLED(uint32_t colorMax, unsigned breathePeriod) {
154 uint8_t redMax = (colorMax & 0xFF0000) >> 16;
155 uint8_t greenMax = (colorMax & 0x00FF00) >> 8;
156 uint8_t blueMax = (colorMax & 0x0000FF);
157
158 // Brightness will rise from 0 to 128, then fall back to 0
159 uint8_t brightness = (m_Counter < 128) ? m_Counter : 255 - m_Counter;
160
161 // Multiply our three colors by the brightness:
162 redMax *= ((float)brightness / 128.0);
163 greenMax *= ((float)brightness / 128.0);
164 blueMax *= ((float)brightness / 128.0);
165 // And turn the LED to that color:
166 setRGB((redMax << 16) | (greenMax << 8) | blueMax);
167
168 // This function relies on the 8-bit, unsigned m_Counter rolling over.
169 m_Counter = (m_Counter+1) % 256;
170 return breathePeriod / 256;
171 }
172
173#else

175 template<typename T>
176 uint32_t beatLED(uint32_t, const T& beat) {
177 const uint8_t cnt = sizeof(beat)/sizeof(beat[0]);
178 setLED((m_Counter % 2 == 0) ? BOARD_LED_BRIGHTNESS : 0);
179 uint32_t next = beat[m_Counter % cnt];
180 m_Counter = (m_Counter+1) % cnt;
181 return next;
182 }

184 uint32_t waveLED(uint32_t, unsigned breathePeriod) {
185 uint32_t brightness = (m_Counter < 128) ? m_Counter : 255 - m_Counter;
186
187 setLED(DIMM(brightness*2));
188
189 // This function relies on the 8-bit, unsigned m_Counter rolling over.
190 m_Counter = (m_Counter+1) % 256;
191 return breathePeriod / 256;
192 }
