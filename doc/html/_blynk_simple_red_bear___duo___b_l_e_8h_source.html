27/*
28 * BLE peripheral preferred connection parameters:
29 * - Minimum connection interval = MIN_CONN_INTERVAL * 1.25 ms, where MIN_CONN_INTERVAL ranges from 0x0006 to 0x0C80
30 * - Maximum connection interval = MAX_CONN_INTERVAL * 1.25 ms, where MAX_CONN_INTERVAL ranges from 0x0006 to 0x0C80
31 * - The SLAVE_LATENCY ranges from 0x0000 to 0x03E8

139 void begin() {

143 //ble.debugLogger(true);
144 // Initialize ble_stack.

146
147 // Register BLE callback functions
148 ble.onConnectedCallback(deviceConnectedCallback);
149 ble.onDisconnectedCallback(deviceDisconnectedCallback);
150 ble.onDataWriteCallback(gattWriteCallback);
151
152 // Add GAP service and characteristics
153 ble.addService(BLE_UUID_GAP);
154 ble.addCharacteristic(BLE_UUID_GAP_CHARACTERISTIC_DEVICE_NAME, ATT_PROPERTY_READ|ATT_PROPERTY_WRITE, (uint8_t*)BLE_DEVICE_NAME, sizeof(BLE_DEVICE_NAME));
155 ble.addCharacteristic(BLE_UUID_GAP_CHARACTERISTIC_APPEARANCE, ATT_PROPERTY_READ, appearance, sizeof(appearance));
156 ble.addCharacteristic(BLE_UUID_GAP_CHARACTERISTIC_PPCP, ATT_PROPERTY_READ, conn_param, sizeof(conn_param));
157
158 // Add GATT service and characteristics
159 ble.addService(BLE_UUID_GATT);
160 ble.addCharacteristic(BLE_UUID_GATT_CHARACTERISTIC_SERVICE_CHANGED, ATT_PROPERTY_INDICATE, change, sizeof(change));
161
162 // Add user defined service and characteristics
163 ble.addService(service1_uuid);
164 character1_handle = ble.addCharacteristicDynamic(service1_tx_uuid, ATT_PROPERTY_WRITE|ATT_PROPERTY_WRITE_WITHOUT_RESPONSE, characteristic1_data, CHARACTERISTIC1_MAX_LEN);
165 character2_handle = ble.addCharacteristicDynamic(service1_rx_uuid, ATT_PROPERTY_NOTIFY, characteristic2_data, CHARACTERISTIC2_MAX_LEN);
166
167 // Set BLE advertising parameters
168 ble.setAdvertisementParams(&adv_params);
169
170 // // Set BLE advertising data
171 ble.setAdvertisementData(sizeof(adv_data), adv_data);
172
173 // BLE peripheral starts advertising now.
174 ble.startAdvertising();
