87 void run() {
88 app_loop();
89 switch (BlynkState::get()) {
90 case MODE_WAIT_CONFIG:
91 case MODE_CONFIGURING: enterConfigMode(); break;
92 case MODE_CONNECTING_NET: enterConnectNet(); break;
93 case MODE_CONNECTING_CLOUD: enterConnectCloud(); break;
94 case MODE_RUNNING: runBlynkWithChecks(); break;
95 case MODE_OTA_UPGRADE: enterError(); break; // TODO: enterOTA
96 case MODE_SWITCH_TO_STA: enterSwitchToSTA(); break;
97 case MODE_RESET_CONFIG: enterResetConfig(); break;
98 default: enterError(); break;
